### Cell 4: Initialize Tokenizer & Preprocessor
```python
MODEL_ID = "google/flan-t5-xl"
tokenizer = AutoTokenizer.from_pretrained(MODEL_ID)

SYSTEM_PROMPT = (
    "You are a medical writer creating a clinical narrative for regulatory submission. "
    "Rewrite this clinical paragraph to flow perfectly using a regulatory medical tone. "
    "Do not add, remove, or change ANY numbers or clinical events."
)

def preprocess(batch):
    # For Phase 3, we teach it to rewrite ML text
    inputs = [f"{SYSTEM_PROMPT}\n\nParagraph:\n{t}\n\nRewritten Narrative:" for t in batch["table_text"]]
    
    model_inputs = tokenizer(inputs, max_length=2048, truncation=True, padding="max_length")
    
    with tokenizer.as_target_tokenizer():
        labels = tokenizer(batch["writeup"], max_length=512, truncation=True, padding="max_length")
        
    model_inputs["labels"] = [
        [(l if l != tokenizer.pad_token_id else -100) for l in label]
        for label in labels["input_ids"]
    ]
    return model_inputs

tokenized_train = splits["train"].map(preprocess, batched=True, remove_columns=splits["train"].column_names)
tokenized_val = splits["test"].map(preprocess, batched=True, remove_columns=splits["test"].column_names)
```

### Cell 5: QLoRA Model Setup (Maximum GPU Efficiency)
*This is where the magic happens. We load a 3 Billion parameter model into just ~4GB of VRAM using 4-bit NF4 quantization, leaving plenty of room for large batch sizes.*
```python
# 4-bit quantization config
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_use_double_quant=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.bfloat16  # H100 loves bfloat16!
)

model = AutoModelForSeq2SeqLM.from_pretrained(
    MODEL_ID,
    quantization_config=bnb_config,
    device_map="auto"
)

# Prepare for LoRA
model = prepare_model_for_kbit_training(model)

config = LoraConfig(
    r=32, 
    lora_alpha=64,
    target_modules=["q", "k", "v", "o", "wi_0", "wi_1", "wo"], # Target all linear layers
    lora_dropout=0.05,
    bias="none",
    task_type=TaskType.SEQ_2_SEQ_LM
)

model = get_peft_model(model, config)
model.print_trainable_parameters()
```

### Cell 6: Metrics Engine
```python
rouge = evaluate.load("rouge")
import numpy as np

def compute_metrics(eval_pred):
    predictions, labels = eval_pred
    decoded_preds = tokenizer.batch_decode(predictions, skip_special_tokens=True)
    
    # Replace -100 in the labels 
    labels = np.where(labels != -100, labels, tokenizer.pad_token_id)
    decoded_labels = tokenizer.batch_decode(labels, skip_special_tokens=True)
    
    result = rouge.compute(predictions=decoded_preds, references=decoded_labels, use_stemmer=True)
    return {k: round(v, 4) for k, v in result.items()}
```

### Cell 7: Launch Training! 🚀
```python
# H100 can handle larger batches. We set 8 per device.
args = Seq2SeqTrainingArguments(
    output_dir="./xl_clinical_adapter",
    evaluation_strategy="epoch",
    learning_rate=2e-4,
    per_device_train_batch_size=8,
    per_device_eval_batch_size=8,
    weight_decay=0.01,
    save_total_limit=3,
    num_train_epochs=15,
    predict_with_generate=True,
    fp16=False,
    bf16=True, # H100 MUST use bf16, not fp16 (prevents overflow issues)
    logging_steps=10,
    optim="paged_adamw_32bit", # crucial for QLoRA stability
)

trainer = Seq2SeqTrainer(
    model=model,
    args=args,
    train_dataset=tokenized_train,
    eval_dataset=tokenized_val,
    tokenizer=tokenizer,
    data_collator=DataCollatorForSeq2Seq(tokenizer, model=model, label_pad_token_id=-100),
    compute_metrics=compute_metrics,
)

trainer.train()

# Save the adapter!
trainer.model.save_pretrained("best_clinical_adapter")
tokenizer.save_pretrained("best_clinical_adapter")
print("✅ Training Complete and Adapter Saved!")
```
